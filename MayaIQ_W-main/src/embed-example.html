üíª JavaScript API

// Get the widget element
const chatWidget = document.querySelector('pingbash-chat');

// Or create programmatically
const widget = PingbashChat.create({
    groupName: 'my-group',
    position: 'bottom-right',
    customColors: { primary: '#ff6b6b' }
});

// Control the widget
chatWidget.open();        // Open the chat
chatWidget.close();       // Close the chat
chatWidget.minimize();    // Minimize the chat
chatWidget.restore();     // Restore from minimized

// Listen to events
chatWidget.addEventListener('pingbash-opened', () => {
    console.log('Chat opened!');
});

chatWidget.addEventListener('pingbash-message', (event) => {
    console.log('New message:', event.detail);
});

// Update configuration
chatWidget.updateConfig({
    theme: 'dark',
    customColors: { primary: '#purple' }
});
üîß Development Note: This example uses a local backend. In production, update the api-url attribute to point to your Pingbash server instance.
